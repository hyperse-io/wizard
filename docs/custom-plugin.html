How to create a custom plugin

A plugin is a function that accepts a Wizard instance and returns a Wizard instance.

Usage

// @filename: @hyperse/wizard/dist/index.d.ts
import { ,  } from '@hyperse/wizard';
 
export const  = () => {
  return ({
    // locale message key
    : 'plugins.newPlugin.name',
    : versionMessages,
    : () => {
      let  = .(
        defineCommand('print', {
          // locale message key
          : 'plugins.newPlugin.command.description',
        }).process(() => {
          ..('hello world');
        })
      );
      return ;
    },
  });
};

  },
})
  .(())
  .();
